function trap(bars) {
  let left  = 0;
  let right = bars.length - 1;
  let leftMax  = 0;
  let rightMax = 0;
  let water = 0;
 
  while (left < right) {
 
    if (bars[left] < bars[right]) {  // left side shorter
      if (bars[left] >= leftMax) {
        leftMax = bars[left];        // new left wall
      } else {
        water += leftMax - bars[left]; // trapped!
      }
      left++;                         // advance left
    } else {                           // right side shorter or equal
      if (bars[right] >= rightMax) {
        rightMax = bars[right];       // new right wall
      } else {
        water += rightMax - bars[right]; // trapped!
      }
      right--;                        // advance right
    }
  }
  return water;
}
