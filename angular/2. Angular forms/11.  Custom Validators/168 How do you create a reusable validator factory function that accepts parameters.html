Reusable Validator Factory Function

A validator factory function accepts a parameter and returns a validator.

TypeScript


minLengthValidator(min: number): ValidatorFn {

  return (control: AbstractControl) => {

    if (control.value.length < min) {

      return { minLength: true };

    }

    return null;

  };

}

Reactive Form


form = this.fb.group({

  username: [

    '',

    [

      this.minLengthValidator(5)

    ]

  ]

});

HTML View


<form [formGroup]="form">

  <input 
    type="text"
    formControlName="username"
    placeholder="Enter username"
  >

  <div 
    class="error"
    *ngIf="form.get('username')?.errors?.['minLength']"
  >

    Minimum 5 characters required

  </div>

</form>

Output Example

Minimum 5 characters required
