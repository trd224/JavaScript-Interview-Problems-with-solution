Important Correction

Correct NgRx Flow

Your old diagram placed Effect after Selector. That is wrong for NgRx. An Effect listens to the same dispatched Action stream. It runs side effects like API calls and then dispatches another Action.

Action
Reducer

Correct mental model: Action is the starting point. Reducer and Effect can both react to that Action. Selector only reads data from Store; Selector does not trigger Effect.

  selector: 'app-counter',

export class Counter {

  count$: Observable<number>

    this.count$ = this.store.select(selectCount)

  increment(){
    this.store.dispatch(increment())
  }

  decrement(){
    this.store.dispatch(decrement())
  }

  reset(){
    this.store.dispatch(reset())
  }

}

Async Flow

Product Flow: Component → Action → Effect → API → Success/Failure → Reducer → Store → Selector → UI

1. Products page dispatches loadProducts() in constructor.
2. ProductEffects listens using ofType(loadProducts).
3. Effect calls ProductService.getAllProduct().
4. Effect dispatches loadProductsSuccess() or loadProductsFailure().
5. productReducer updates products, loading and error.
6. Product selectors read data and component displays it using selectSignal().

import { inject, Injectable } from '@angular/core';
import { Actions, createEffect, ofType } from '@ngrx/effects';
import { ProductService } from '../../services/product.service';
import * as ProductActions from './product.actions';
import { catchError, map, mergeMap, of } from 'rxjs';

@Injectable()
export class ProductEffects {

  private actions$ = inject(Actions);
  private productService = inject(ProductService);

  loadProducts$ = createEffect(() =>
    this.actions$.pipe(
      ofType(ProductActions.loadProducts),
      mergeMap(() =>
        this.productService.getAllProduct().pipe(
          map(products =>
            ProductActions.loadProductsSuccess({ products })
          ),
          catchError(error =>
            of(ProductActions.loadProductsFailure({ error }))
          )
        )
      )
    )
  );
}

@Injectable({
  providedIn: 'root',
})
export class ProductService {
   http = inject(HttpClient);

  getAllProduct(): Observable<IProduct[]>{
    return this.http.get<IProduct[]>('https://fakestoreapi.com/products').pipe(
      map((products) => {
        return products.map((product) => {
          return {...product, quantity: 1}
        })
      })
    )
  }
}

import { createReducer, on } from "@ngrx/store";
import { IProduct } from "../../models/product.interface";
import { addToCart, decrement, increment, removeItem } from "./cart.actions";

export interface CartState {
    products: IProduct[],
    subTotal: number;
    deliveryPrice: number;
    discount: number;
    totalPrice: number;
}

export const initialState: CartState = {
    products: [],
    subTotal: 0,
    deliveryPrice: 50,   // example fixed delivery charge
    discount: 0,
    totalPrice: 0
}

export function calculateSubTotal(products: IProduct[]){
    return products.reduce((total, product) => total + (product.price * product.quantity), 0);
}

export function calculateTotal(subTotal: number,deliveryPrice: number,discount: number) {
  return subTotal + deliveryPrice - discount;
}

export const cartReducer = createReducer(
    initialState,
    on(addToCart, (state, {product}) => {
        const updatedProducts = [...state.products, product];
        const subTotal = calculateSubTotal(updatedProducts);
        const totalPrice = calculateTotal(subTotal, state.deliveryPrice, state.discount);
        return {
            ...state, 
            products: updatedProducts,
            subTotal: subTotal,
            totalPrice: totalPrice
        }
    }),

   on(increment, (state, {itemId}) => {
    const updatedProducts = state.products.map((product) => {
        return product.id === itemId ? {...product, quantity: product.quantity + 1} : product
    })
     const subTotal = calculateSubTotal(updatedProducts);
     const totalPrice = calculateTotal(subTotal, state.deliveryPrice, state.discount);

    return {
        ...state,
        products: updatedProducts,
        subTotal: subTotal,
        totalPrice: totalPrice
    }

   }),

   on(decrement, (state, {itemId}) => {
    const updatedProducts = state.products.map((product) => {
        return product.id === itemId ? {...product, quantity: product.quantity - 1} : product
    })
    const subTotal = calculateSubTotal(updatedProducts);
    const totalPrice = calculateTotal(subTotal, state.deliveryPrice, state.discount);
    return {
        ...state,
        products: updatedProducts,
        subTotal: subTotal,
        totalPrice: totalPrice
    }
   }),

   on(removeItem, (state, {itemId}) => {
    const updatedProducts = state.products.filter((product) => {
        return product.id !== itemId 
    })
    const subTotal = calculateSubTotal(updatedProducts);
    const totalPrice = calculateTotal(subTotal, state.deliveryPrice, state.discount);
    return {
        ...state,
        products: updatedProducts,
        subTotal: subTotal,
        totalPrice: totalPrice
    }
   })
)

Remember

NgRx File Responsibility Map

actions.ts
Only declares events: loadProducts, addToCart, increment.
reducer.ts
Only changes state immutably. No API call here.
effects.ts
Only for side effects like HTTP. It listens to actions and returns new actions.
selectors.ts
Only reads/derives state from Store. It does not modify state.
app.config.ts
Registers provideStore(), provideState(), and provideEffects().
component.ts
Dispatches actions and reads selectors using store.select() or store.selectSignal().
