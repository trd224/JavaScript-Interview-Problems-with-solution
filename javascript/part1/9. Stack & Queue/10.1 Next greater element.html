function nextGreaterElement(arr) {
  let stack  = [];
  let result = new Array(arr.length);
 
  for (let i = arr.length - 1; i >= 0; i--) {
 
    // Remove smaller elements (they can never be NGE)
    while (stack.length && stack[stack.length-1] <= arr[i]) {
      stack.pop();
    }
 
    // Stack empty → no greater element exists to the right
    result[i] = stack.length === 0 ? -1 : stack[stack.length-1];
 
    // Push current value as candidate for elements to its left
    stack.push(arr[i]);
  }
 
  return result;
}
