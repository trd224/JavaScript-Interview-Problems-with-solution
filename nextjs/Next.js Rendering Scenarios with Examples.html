Next.js Rendering Scenarios with Examples

1. (SSG Page) SSG Page + Static Server Component 

// app/page.tsx
import About from './About';

  return <About />;
}

2. SSG Page (API Data)

// app/page.tsx
import ProductList from './ProductList';

export default async function Page() {
  const products = await fetch(
    'https://api.example.com/products'
  ).then(res => res.json());

  return <ProductList products={products} />;
}
// app/ProductList.tsx
export default function ProductList({ products }) {
  return (
    <ul>
      {products.map(product => (
        <li key={product.id}>
          {product.name}
        </li>
      ))}
    </ul>
  );
}
Result: SSG Hydration: No HTML Generated: Build Time Regenerated: Never View Source: Shows Products SEO: Excellent Note: Products are fetched only during build time.

4. SSG Page + Client Component + API Fetch

// app/page.tsx
import RecentlyViewed from './RecentlyViewed';

export default function Page() {
  return <RecentlyViewed />;
}
// app/RecentlyViewed.tsx
'use client';

import { useEffect, useState } from 'react';

export default function RecentlyViewed() {
  const [products, setProducts] = useState([]);

  useEffect(() => {
    fetch('/api/recently-viewed')
      .then(res => res.json())
      .then(setProducts);
  }, []);

  return products.map(product => <p key={product.id}>{product.name}</p>);
}
Result: SSG + Hydration Hydration: Yes HTML Generated: Build Time API Call: Browser Regenerated: Never View Source: Does NOT show fetched products SEO: Poor for fetched products

5. SSG → SSR (cookies)

// app/page.tsx
import UserName from './UserName';

export default function Page() {
  return <UserName />;
}
// app/UserName.tsx
import { cookies } from 'next/headers';

export default async function UserName() {
  const cookieStore = await cookies();
  const name = cookieStore.get('name')?.value;

  return <h1>Hello {name}</h1>;
}
Result: SSR / Dynamic Hydration: No HTML Generated: Request Time Build Time Generation: Not Possible View Source: Shows User Name SEO: Good

7. SSG → SSR (connection)

// app/page.tsx
import CurrentTime from './CurrentTime';

export default function Page() {
  return <CurrentTime />;
}
// app/CurrentTime.tsx
import { connection } from 'next/server';

export default async function CurrentTime() {
  await connection();

  return <p>{new Date().toISOString()}</p>;
}
Result: SSR / Dynamic Hydration: No HTML Generated: Request Time Build Time Generation: Not Possible View Source: Shows Current Time SEO: Good
Note: Without connection, Next.js may prerender this page as SSG, so the time can become fixed at build time.
Next.js may decide: This page has no cookies() No headers() No searchParams No request data I can prerender it. (SSG)

8. ISR Page (API Data)

// app/page.tsx
import ProductList from './ProductList';

export default async function Page() {
  const products = await fetch('https://api.example.com/products', {
    next: { revalidate: 60 }
  }).then(res => res.json());

  return <ProductList products={products} />;
}
// app/ProductList.tsx
export default function ProductList({ products }) {
  return (
    <ul>
      {products.map(product => <li key={product.id}>{product.name}</li>)}
    </ul>
  );
}
Result: ISR Hydration: No HTML Generated: Build Time Regenerated: Every 60 seconds View Source: Shows Products SEO: Excellent

11. ISR Page + Client Component + API Fetch

// app/page.tsx
import CartCount from './CartCount';

export const revalidate = 60;

export default function Page() {
  return <CartCount />;
}
// app/CartCount.tsx
'use client';

import { useEffect, useState } from 'react';

export default function CartCount() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    fetch('/api/cart-count')
      .then(res => res.json())
      .then(data => setCount(data.count));
  }, []);

  return <span>Cart: {count}</span>;
}

16. SSR Page + Client Component + API Fetch

// app/dashboard/page.tsx
import Notifications from './Notifications';

export const dynamic = 'force-dynamic';

export default function Page() {
  return <Notifications />;
}
// app/dashboard/Notifications.tsx
'use client';

import { useEffect, useState } from 'react';

export default function Notifications() {
  const [items, setItems] = useState([]);

  useEffect(() => {
    fetch('/api/notifications')
      .then(res => res.json())
      .then(setItems);
  }, []);

  return items.map(item => <p key={item.id}>{item.message}</p>);
}
Result: SSR + Hydration Hydration: Yes HTML Generated: Request Time API Call: Browser View Source: Does NOT show fetched notifications SEO: Poor for fetched notifications

Super Simple Rule

Server Component = No Hydration Client Component = Hydration Static data = SSG Public changing data = ISR cookies / headers / connection = SSR Client API fetch = Data comes after hydration
