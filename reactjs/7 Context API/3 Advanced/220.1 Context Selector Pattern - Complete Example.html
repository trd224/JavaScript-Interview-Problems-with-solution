220. Context Selector Pattern - Complete Example

The Context Selector Pattern lets components subscribe only to the context value they need, preventing unnecessary re-renders.

Install

npm install use-context-selector

Folder Structure

src/
├── App.js
├── AppContext.js
├── UserName.js
├── ThemeButton.js
├── CartCount.js
└── index.js

AppContext.js

import { createContext } from "use-context-selector";
import { useState } from "react";

export const AppContext=createContext(null);

export function AppProvider({ children }){
 const [user,setUser]=useState({name:"Rizwan",age:25});
 const [theme,setTheme]=useState("Light");
 const [cart,setCart]=useState([]);
 
 return (
    <AppContext.Provider value={{user,setUser,theme,setTheme,cart,setCart}}>
        {children}
    </AppContext.Provider>);
}

UserName.js

const user=useContextSelector(AppContext,v=>v.user);
return <h2>User : {user.name}</h2>;

ThemeButton.js

const theme=useContextSelector(AppContext,v=>v.theme);
const setTheme=useContextSelector(AppContext,v=>v.setTheme);
<button onClick={()=>setTheme(t=>t==="Light"?"Dark":"Light")}>Change Theme</button>

CartCount.js

const cart=useContextSelector(AppContext,v=>v.cart);
const setCart=useContextSelector(AppContext,v=>v.setCart);
<button onClick={()=>setCart(c=>[...c,{id:Date.now(),name:"Laptop"}])}>Add Item</button>

App.js

function App(){
 return (
    <AppProvider>
        <UserName/>
        <ThemeButton/>
        <CartCount/>
    </AppProvider>);
}

index.js

import ReactDOM from "react-dom/client";
ReactDOM.createRoot(document.getElementById("root")).render(<App />);

Render Behavior

ActionComponent Re-rendered
Change ThemeThemeButton only
Add ItemCartCount only
Update UserUserName only
Each component subscribes only to the slice of context it needs, improving performance.
