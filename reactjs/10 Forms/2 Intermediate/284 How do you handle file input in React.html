284. How do you handle file input in React?

One-Line Definition

File input is handled using <input type="file" /> and accessing selected files through event.target.files.

How It Works

User Selects File
↓
onChange Event
↓
event.target.files
↓
Get File Object
↓
Preview / Upload

Basic Example

File: src/components/FileUpload.jsx
import { useState } from 'react';

function FileUpload() {
  const [file, setFile] = useState(null);

  return (
    <>
      <input type="file" onChange={e => setFile(e.target.files[0])} />
      {file && <h3>{file.name}</h3>}
    </>
  );
}

export default FileUpload;

Image Preview Example

File: src/components/ImageUpload.jsx
import { useState } from 'react';

function ImageUpload() {
  const [preview, setPreview] = useState('');

  function handleFileChange(e) {
    const file = e.target.files[0];
    if (!file) return;

    setPreview(URL.createObjectURL(file));
  }

  return (
    <>
      <input type="file" accept="image/*" onChange={handleFileChange} />
      {preview && <img src={preview} alt="Preview" width="200" />}
    </>
  );
}

export default ImageUpload;

Upload File To API

File: src/components/FileUpload.jsx
import axios from 'axios';
import { useState } from 'react';

function FileUpload() {
  const [file, setFile] = useState(null);

  async function uploadFile() {
    const formData = new FormData();
    formData.append('file', file);
    await axios.post('/api/upload', formData);
  }

  return (
    <>
      <input type="file" onChange={e => setFile(e.target.files[0])} />
      <button onClick={uploadFile}>Upload</button>
    </>
  );
}

export default FileUpload;

Why Use FormData?

File Object
↓
FormData
↓
multipart/form-data
↓
Server

Files cannot be sent as normal JSON.

Common Mistake

const [file, setFile] = useState('');

<input value={file} type="file" />
File inputs cannot be controlled. Never use the value prop on a file input.

File Input vs Text Input

Text Input File Input
event.target.value event.target.files
String Value File Object
Can Be Controlled Cannot Be Controlled

Interview Answer

File input is handled in React using an input element with type="file". Selected files are accessed through event.target.files, stored in state if needed, previewed using URL.createObjectURL(), and uploaded using FormData.

One-Line Interview Answer

Use <input type="file" /> and access selected files using event.target.files.
